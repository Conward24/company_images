  <script>

    // Generate or retrieve a unique user ID from localStorage

    if (!localStorage.getItem("luaUserID")) {

      localStorage.setItem("luaUserID", "user_" + Math.random().toString(36).substr(2, 9));

    }

    const userID = localStorage.getItem("luaUserID");

    const chatLog = document.getElementById("lua-chat-log");


    // Function to send predefined prompts

    function sendPrompt(promptText) {

      const userInput = document.getElementById("lua-user-input");

      userInput.value = promptText;

      document.getElementById("lua-send-button").click();

    }

    

    // Function to selectively bold sentences that end with a question mark.

    function formatResponse(text) {

      const sentences = text.match(/[^.!?]+[.!?]+|[^.!?]+$/g);

      if (!sentences) return text;

      return sentences.map(sentence => {

        sentence = sentence.trim();

        if (sentence.endsWith('?')) {

          return `<strong>${sentence}</strong>`;

        }

        return sentence;

      }).join(' ');

    }

    

    // Event listener for Send button

    document.getElementById("lua-send-button").addEventListener("click", async () => {

      const userInputVal = document.getElementById("lua-user-input").value;

      if (!userInputVal.trim()) return;

      

      // Append user's message

      const userMessage = document.createElement("div");

      userMessage.className = "lua-message user";

      userMessage.innerHTML = `<div class="lua-message-content">${userInputVal}</div><div class="lua-message-icon">💬</div>`;

      chatLog.appendChild(userMessage);

      userMessage.scrollIntoView({ block: "start", behavior: "smooth" });

      

      // Clear input field

      document.getElementById("lua-user-input").value = "";

      

      // Append thinking animation

      const thinkingMessage = document.createElement("div");

      thinkingMessage.className = "lua-message thinking";

      thinkingMessage.innerHTML = `<div class="lua-thinking-animation"><img class="static-lotus" src="https://raw.githubusercontent.com/Conward24/flask-chatbot/refs/heads/main/the_lotus_only2.svg" alt="Static Lotus"><img class="animated-orb" src="https://raw.githubusercontent.com/Conward24/flask-chatbot/refs/heads/main/orb_only2.svg" alt="Animated Orb"></div>`;

      chatLog.appendChild(thinkingMessage);

      

      try {

        const response = await fetch("https://conward2.pythonanywhere.com/query", {

          method: "POST",

          headers: { "Content-Type": "application/json" },

          body: JSON.stringify({ query: userInputVal, user_id: userID })

        });

        const data = await response.json();

        const mainResponse = (data.response || "I'm sorry, I couldn't process your query.").trim();

        

        // Remove the thinking indicator

        chatLog.removeChild(thinkingMessage);

        

        // Format and append bot response

        const formattedResponse = formatResponse(mainResponse);

        const botMessage = document.createElement("div");

        botMessage.className = "lua-message bot";

        botMessage.innerHTML = `<div class="lua-message-icon"><img src="https://raw.githubusercontent.com/Conward24/flask-chatbot/main/thelotuspng.png" alt="Bot Icon"></div><div><div class="lua-message-content">${formattedResponse}</div><div class="lua-feedback"><button class="like" data-tooltip="Good response"><i class="far fa-thumbs-up"></i></button><button class="dislike" data-tooltip="Bad response"><i class="far fa-thumbs-down"></i></button></div></div>`;

        chatLog.appendChild(botMessage);

        

        // Append follow-up links, if any

        const followUpLinks = data.follow_up_links || [];

        followUpLinks.forEach(link => {

          const linkMessage = document.createElement("div");

          linkMessage.className = "lua-message bot";

          linkMessage.innerHTML = `<div class="lua-message-icon"><img src="https://raw.githubusercontent.com/Conward24/flask-chatbot/main/thelotuspng.png" alt="Bot Icon"></div><div><div class="lua-message-content">For more info: <a href="${link.url.trim()}" target="_blank" style="color: #2563EB; text-decoration: underline;">${link.url.trim()}</a></div></div>`;

          chatLog.appendChild(linkMessage);

        });

      } catch (error) {

        if (chatLog.contains(thinkingMessage)) {

          chatLog.removeChild(thinkingMessage);

        }

        const botMessage = document.createElement("div");

        botMessage.className = "lua-message bot";

        botMessage.innerHTML = `<div class="lua-message-icon"><img src="https://raw.githubusercontent.com/Conward24/flask-chatbot/main/thelotuspng.png" alt="Bot Icon"></div><div><div class="lua-message-content">Error: ${error.message}</div><div class="lua-feedback"><button class="like" data-tooltip="Good response"><i class="far fa-thumbs-up"></i></button><button class="dislike" data-tooltip="Bad response"><i class="far fa-thumbs-down"></i></button></div></div>`;

        chatLog.appendChild(botMessage);

      }

    });

    

    // Feedback buttons event delegation

    document.getElementById("lua-chat-log").addEventListener("click", (event) => {

      if (event.target.closest(".like")) {

        event.target.closest("button").style.color = "#6b7280";

      } else if (event.target.closest(".dislike")) {

        event.target.closest("button").style.color = "#6b7280";

      }

    });
